
model{
  for(i in 1:n){
    p[i]<-1/(1+exp(-logitP[i]))
    logitP[i]<-intercept[i]+date[i]*slope[i]
    intercept[i]<-overallInt+year[i]*startDateSlope
    slope[i]<-overallSlope+year[i]*durationSlope
    spawned[i]~dbern(p[i])
  }

  overallSlope~dnorm(0,0.01)
  overallInt~dnorm(0,0.01)
  startDateSlope~dnorm(0,0.01)
  durationSlope~dnorm(0,0.01)
}
